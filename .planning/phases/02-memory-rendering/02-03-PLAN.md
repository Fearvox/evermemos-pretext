```ts
import { MemoryItemSchema, type MemoryItem, type MemoryType } from '../types'
import { type PaginatedResponse } from '../types/pagination'

const USE_MOCK = process.env.NEXT_PUBLIC_USE_MOCK !== 'false'
const API_BASE = process.env.NEXT_PUBLIC_API_BASE ?? 'http://localhost:8000'
const PAGE_SIZE = 50

export type FetchMemoriesParams = {
  cursor?: string | null
  type?: MemoryType | null
  query?: string | null
  tags?: string[] | null
  dateFrom?: string | null
  dateTo?: string | null
  signal?: AbortSignal
}

async function fetchMock(params: FetchMemoriesParams): Promise<PaginatedResponse<MemoryItem>> {
  // Dynamic import to avoid bundling mock data in production
  const { mockAllMemories } = await import('../../hub/src/lib/mock')

  let items = [...mockAllMemories]

  // Filter by type
  if (params.type) {
    items = items.filter((item) => item.type === params.type)
  }

  // Filter by tags (intersection: item must have ALL selected tags)
  if (params.tags && params.tags.length > 0) {
    items = items.filter((item) => params.tags!.every((tag) => item.tags.includes(tag)))
  }

  // Filter by query (simple case-insensitive search)
  if (params.query) {
    const q = params.query.toLowerCase()
    items = items.filter((item) => {
      if (item.type === 'episode') {
        return item.title.toLowerCase().includes(q) ||
          item.turns.some((t) => t.content.toLowerCase().includes(q))
      }
      if (item.type === 'profile') {
        return item.name.toLowerCase().includes(q) ||
          item.description.toLowerCase().includes(q)
      }
      return item.subject.toLowerCase().includes(q) ||
        item.content.toLowerCase().includes(q)
    })
  }

  // Filter by date range
  if (params.dateFrom) {
    const from = new Date(params.dateFrom).getTime()
    items = items.filter((item) => new Date(item.createdAt).getTime() >= from)
  }
  if (params.dateTo) {
    const to = new Date(params.dateTo).getTime()
    items = items.filter((item) => new Date(item.createdAt).getTime() <= to)
  }

  // Cursor-based pagination (cursor = index as string)
  const startIndex = params.cursor ? parseInt(params.cursor, 10) : 0
  const page = items.slice(startIndex, startIndex + PAGE_SIZE)
  const nextCursor = startIndex + PAGE_SIZE < items.length
    ? String(startIndex + PAGE_SIZE)
    : null

  return {
    data: page,
    cursor: nextCursor,
    hasMore: nextCursor !== null,
    total: items.length,
  }
}

async function fetchLive(
  endpoint: string,
  params: Record<string, string | undefined>,
  signal?: AbortSignal,
): Promise<Response> {
  const url = new URL(endpoint, API_BASE)
  for (const [key, value] of Object.entries(params)) {
    if (value != null) url.searchParams.set(key, value)
  }
  return fetch(url.toString(), { signal })
}

export async function fetchMemories(
  params: FetchMemoriesParams = {},
): Promise<PaginatedResponse<MemoryItem>> {
  if (USE_MOCK) return fetchMock(params)

  const res = await fetchLive('/api/v1/memories', {
    cursor: params.cursor ?? undefined,
    type: params.type ?? undefined,
    q: params.query ?? undefined,
    tags: params.tags?.join(',') ?? undefined,
    date_from: params.dateFrom ?? undefined,
    date_to: params.dateTo ?? undefined,
  }, params.signal)

  const json = await res.json()
  return json as PaginatedResponse<MemoryItem>
}

export async function fetchMemoryById(
  id: string,
  signal?: AbortSignal,
): Promise<MemoryItem | null> {
  if (USE_MOCK) {
    const { mockAllMemories } = await import('../../hub/src/lib/mock')
    return mockAllMemories.find((item) => item.id === id) ?? null
  }

  const res = await fetchLive(`/api/v1/memories/${id}`, {}, signal)
  if (!res.ok) return null
  const json = await res.json()
  return json as MemoryItem
}
```
