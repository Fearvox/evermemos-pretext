Create `hub/src/components/memory/SearchBar.tsx` as a `'use client'` component:

1. Props: `defaultValue?: string`, `onSearch: (query: string) => void`, `resultCount?: number`
2. Use `useState` for local input value, `useRef` for AbortController and debounce timer
3. Debounce pattern:
   ```tsx
   const timerRef = useRef<ReturnType<typeof setTimeout>>()
   const controllerRef = useRef<AbortController>()

   function handleChange(value: string) {
     setInputValue(value)
     clearTimeout(timerRef.current)
     controllerRef.current?.abort()
     timerRef.current = setTimeout(() => {
       controllerRef.current = new AbortController()
       onSearch(value)
     }, 300)
   }
   ```
4. Render shadcn `Input` with:
   - Placeholder: "Search memories..."
   - Lucide `Search` icon as prefix (absolute positioned left)
   - `className="bg-[var(--surface-secondary)] border-[var(--border)] text-[var(--text-primary)] placeholder:text-[var(--text-muted)] focus-visible:ring-[var(--accent)] h-10 pl-10"`
5. Below input, if `resultCount` is defined and query is non-empty:
   - Show `{resultCount} memories` in 12px/500 `text-[var(--text-muted)]`
6. Sync with URL: use `useSearchParams` and `useRouter` from `next/navigation`:
   ```tsx
   const router = useRouter()
   const searchParams = useSearchParams()

   function onSearch(query: string) {
     const params = new URLSearchParams(searchParams.toString())
     if (query) params.set('q', query)
     else params.delete('q')
     router.push(`/memories?${params.toString()}`)
   }
   ```
</action>

Create `hub/src/components/memory/TagFilter.tsx` as a `'use client'` component:

1. Props: `availableTags: string[]` — the deduplicated list of tags extracted from the current memory data set (passed down from MemoryListLoader)
2. Reads currently selected tags from URL via `useSearchParams`: `?tags=tag1,tag2`
3. Renders as a row of small toggle pills below the type filter
4. Each pill: clicking toggles the tag on/off in the URL search params
5. Styling:
   - Inactive: `bg-[var(--surface-tertiary)] text-[var(--text-muted)]`
   - Active: `bg-[var(--accent)]/20 text-[var(--text-primary)]`
   - Common: `text-xs font-medium px-2 py-0.5 rounded-full cursor-pointer transition-colors duration-150`
6. Sync with URL via `useSearchParams` + `useRouter`:
   - Store as `?tags=planning,multilingual` (comma-separated)
   - Empty means no tag filter (show all)

```tsx
'use client'

import { useRouter, useSearchParams } from 'next/navigation'
import { cn } from '@/lib/utils'

interface TagFilterProps {
  availableTags: string[]
}

export function TagFilter({ availableTags }: TagFilterProps) {
  const router = useRouter()
  const searchParams = useSearchParams()
  const selectedTags = searchParams.get('tags')?.split(',').filter(Boolean) ?? []

  function toggleTag(tag: string) {
    const params = new URLSearchParams(searchParams.toString())
    const current = params.get('tags')?.split(',').filter(Boolean) ?? []
    const next = current.includes(tag)
      ? current.filter((t) => t !== tag)
      : [...current, tag]
    if (next.length > 0) params.set('tags', next.join(','))
    else params.delete('tags')
    router.push(`/memories?${params.toString()}`)
  }

  if (availableTags.length === 0) return null

  return (
    <div className="flex flex-wrap gap-1.5 mt-2">
      {availableTags.map((tag) => {
        const isActive = selectedTags.includes(tag)
        return (
          <button
            key={tag}
            onClick={() => toggleTag(tag)}
            className={cn(
              'text-xs font-medium px-2 py-0.5 rounded-full cursor-pointer transition-colors duration-150',
              isActive
                ? 'bg-[var(--accent)]/20 text-[var(--text-primary)]'
                : 'bg-[var(--surface-tertiary)] text-[var(--text-muted)]',
            )}
          >
            {tag}
          </button>
        )
      })}
    </div>
  )
}
```
